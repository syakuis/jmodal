jQuery Syaku Modal

기본 옵션

default = {
	'backgroundColor': '#000', // 배경색
	'buttonClose': false, // 닫기 버튼 사용여부
	'opacity': 0.75, // 배경 투명도
	'zIndex': 100, // 모달 레이어번호
	'beforeOpen': null, // 열기 전 이벤트
	'afterOpen': null, // 열기 후 이벤트
	'beforeClose': null, // 닫기 전 이벤트
	'afterClose': null, // 닫기 후 이벤트
	'class': null, // 직접 클래스 추가.
	'width': null // 모달 고정된 크기
}

});
