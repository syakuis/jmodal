jQuery Syaku Modal

기본 옵션

default = {
	'backgroundColor': '#000', // 배경색
	'opacity': 0.75, // 배경 투명도
	'zIndex': 100, // 모달 레이어번호
	'beforeOpen': null, // 열기 전 이벤트
	'afterOpen': null, // 열기 후 이벤트
	'beforeClose': null, // 닫기 전 이벤트
	'afterClose': null, // 닫기 후 이벤트
	'class': null // 직접 클래스 추가.
}

모달 열기

esc 키로 닫을 수 있다.
var modal = $('#modal1').jmodal();
$('#btn1').click(function() {
	modal.open();
});

모달 닫기

esc 키로 닫을 수 있다. 
var modal2 = $('#modal2').jmodal();
$('#btn2').click(function() {
	modal2.open();
});
$('#close2').click(function() {
	modal2.close();
});
